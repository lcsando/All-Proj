# List of columns to apply IQR-based outlier handling
columns = ['Principal']

# Loop through each column
for col in columns:
    # Calculate Q1 (25th percentile) and Q3 (75th percentile)
    Q1 = df[col].quantile(0.25)
    Q3 = df[col].quantile(0.75)

    # Calculate the Interquartile Range (IQR)
    IQR = Q3 - Q1

    # Calculate lower and upper bounds for outliers
    lower = Q1 - 1.5 * IQR
    upper = Q3 + 1.5 * IQR

    # Cap the outliers: values above upper bound will be set to the upper bound
    # Optionally, you can handle lower bounds similarly if needed
    df.loc[df[col] > upper, col] = upper
    df.loc[df[col] < lower, col] = lower  # To cap lower outliers

# Dictionary to store the performance metrics of each model
performance = {}
accuracy_comparison = {}

# Train and evaluate each model
def train_eval_model(model,model_name):
    model.fit(X_train_scaled, y_train)
    y_pred = model.predict(X_test_scaled)

    y_train_pred = model.predict(X_train_scaled)

    # Evaluate the model on test data
    accuracy = accuracy_score(y_test, y_pred)
    precision = precision_score(y_test, y_pred)
    recall = recall_score(y_test, y_pred)
    f1 = f1_score(y_test, y_pred)

    # Calculate train and test accuracy for overfitting detection
    train_accuracy = accuracy_score(y_train, y_train_pred)
    test_accuracy = accuracy_score(y_test, y_pred)
    accuracy_comparison[model_name] = {
        "Train Accuracy": train_accuracy,
        "Test Accuracy": test_accuracy
    }

    # Store performance metrics
    performance[model_name] = {
        "Accuracy": accuracy,
        "Precision": precision,
        "Recall": recall,
        "F1 Score": f1
    }

    # Confusion Matrix
    cm = confusion_matrix(y_test, y_pred)
    plt.figure(figsize=(6, 4))
    sns.heatmap(cm, annot=True, fmt='d', cmap='Blues', xticklabels=["Paid Off", "Collection"], yticklabels=["Paid Off", "Collection"])
    plt.title(f'Confusion Matrix for {model_name}')
    plt.xlabel('Predicted')
    plt.ylabel('Actual')
    plt.show()

    # Print classification report
    print("Confusion Matrix:\n", confusion_matrix(y_test, y_pred))
    print(f"Model: {model_name}")
    print("Classification Report:\n", classification_report(y_test, y_pred))

    result1.append(accuracy)
    result2.append(precision)
    result3.append(recall)
    result4.append(f1)


def comparission_models() :
    performance_df = pd.DataFrame(performance).T
    print("Performance Comparison:\n", performance_df)

def train_test_accuracy_all() :
    #Plot train vs test accuracy to check for overfitting
    accuracy_df = pd.DataFrame(accuracy_comparison).T
    accuracy_df.plot(kind='bar', figsize=(10, 6))
    plt.title('Train vs Test Accuracy Comparison')
    plt.xlabel('Models')
    plt.ylabel('Accuracy')
    plt.xticks(rotation=45)
    plt.show()
